    # Input-Number 属性

    参数 说明 类型 可选值 默认值
    value/v-model 绑定值 string/number -- --
    disabled 是否禁用 boolean -- false
    max 最大值 number -- --
    min 最小值 number -- --
    size 尺寸 string large/medium/small/mini medium
    step 步长 number -- 1
    step-strictly 是否只能输入 step 的倍数 boolean -- false
    precision 数值精度 number -- --
